\frame{

    Sugar for deleting branches:
    \begin{itemize}
        \item git push origin :goner
        \item git push origin --delete goner
    \end{itemize}
    Other stuff:
    \begin{itemize}
        \item New diff algorithm available called "patience"
        \item git merge --edit \# instead of --nocommit
        \item git submodule update --init --recursive
    \end{itemize}
}

\frame{
    \frametitle{What is libgit2, seriously?}
    \begin{itemize}
        \item Reimplementation of Git
        \item Like a kid sister, not a fork or replacement
        \item Fixes issues that are baked into Git 1.x being for the CLI, not a library
        \item Portable, supports Windows natively
        \item Lighter footprint, better for embedded/mobile apps
    \end{itemize}
}

\frame{

        \item Look inside repos
        \item Create commits, search through commits
